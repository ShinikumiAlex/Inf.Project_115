Go

Подробная информация о языке программирования Go: история, особенности, преимущества, недостатки.

Примеры кода


package main

import "fmt"

func main() {
    fmt.Println("Hello, World!")
}

func greet(name string) {
    fmt.Printf("Привет, %s!\n", name)
}

func init() {
	greet("Go")
}
